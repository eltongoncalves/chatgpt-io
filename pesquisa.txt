import com.fasterxml.jackson.annotation.JsonProperty;
import lombok.Data;

import java.util.List;

@Data
public class ContratoCredito {
    @JsonProperty("nomeTitular")
    private String nomeTitular;

    @JsonProperty("cpfCnpj")
    private String cpfCnpj;

    @JsonProperty("dataEmissao")
    private String dataEmissao;

    @JsonProperty("limiteTotal")
    private Double limiteTotal;

    @JsonProperty("limiteDisponivel")
    private Double limiteDisponivel;

    @JsonProperty("limiteUtilizado")
    private Double limiteUtilizado;

    @JsonProperty("caracteristicaCredito")
    private CaracteristicaCredito caracteristicaCredito;

    @JsonProperty("parcelamentoFatura")
    private List<Parcela> parcelamentoFatura;

    @JsonProperty("parcelamentoAutomatico")
    private List<Parcela> parcelamentoAutomatico;

    @JsonProperty("saldoRotativo")
    private List<SaldoRotativo> saldoRotativo;
}
