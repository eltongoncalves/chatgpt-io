    private static String transformarEmJson(String message) {
        if (Objects.nonNull(message) && message.contains("\\")) {
            message = message
                    .replaceFirst("\"", "")
                    .replace("\\\"", "\"")
                    .replace("\\\\", "\\");

            if (!message.isEmpty()) {
                message = message.substring(0, message.length() - 1);
            }
            logger.info("Mensagem tratada com sucesso");
        }
        return message;
    }
