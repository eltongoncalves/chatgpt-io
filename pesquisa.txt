private static String transformarEmJson(String message) {
    if (Objects.nonNull(message) && message.contains("\\")) {
        message = message
                .replaceFirst("^\"", "")     // Remove aspas iniciais, se presente
                .replace("\\\"", "\"")       // Substitui aspas com escape por aspas simples
                .replace("\\\\", "\\");      // Substitui barras invertidas duplas por uma única

        // Remove o último caractere, se não estiver vazio
        if (!message.isEmpty() && message.endsWith("\"")) {
            message = message.substring(0, message.length() - 1);
        }

        logger.info("Mensagem tratada com sucesso");
    }
    return message;
}
